472         * in the problem (via its path) with the given problem
473         * attributes
474         * @param problem the problem to create a marker from
475         */
476        private void createMarkerForProblem(String type, IApiProblem problem) {

479                        return;

495                                                        IMarker.SOURCE_ID,
496                                                        IApiMarkerConstants.MARKER_ATTR_PROBLEM_ID},

499                                                        new Integer(ApiPlugin.getDefault().getSeverityLevel(ApiProblemFactory.getProblemSeverityId(problem), this.fCurrentProject)),
500                                                        new Integer(line),
501                                                        new Integer(problem.getCharStart()),
502                                                        new Integer(problem.getCharEnd()),
503                                                        ApiAnalysisBuilder.SOURCE,

505                                        }
506                                );

636        /**
637         * Returns an array of type names, and cleans up markers for the specified resource
638         * @param alltypes the listing of {@link IFile}s to get qualified names from
639         * @param changedtypes the listing of {@link IFile}s that have actually changed (from the {@link IResourceDelta}
640         * @param tnames the list to collect all type names into (including inner member names)
641         * @param cnames the list to collect the changed type names into
642         * @param monitor
643         */
644        private void collectAllQualifiedNames(final HashSet alltypes, List tnames, List cnames, final IProgressMonitor monitor) {
645                IType[] types = null;
646                IFile file = null;
647                for (Iterator iterator = alltypes.iterator(); iterator.hasNext(); ) {
648                        file = (IFile) iterator.next();
649                        ICompilationUnit unit = (ICompilationUnit) JavaCore.create(file);
650                        if(!unit.exists()) {
651                                continue;
652                        }
653                        IType type = unit.findPrimaryType();
654                        if(type == null) {
655                                continue;
656                        }
657                        updateMonitor(monitor, 0);
658                        cleanupUnsupportedTagMarkers(file);
659                        updateMonitor(monitor, 0);
660                        cleanupCompatibiltiyMarkers(file);
661                        updateMonitor(monitor, 0);
662                        cnames.add(type.getFullyQualifiedName());
663                        try {
664                                cleanupUsageMarkers(file);
665                                updateMonitor(monitor, 0);
666                                types = unit.getAllTypes();
667                                for (int i = 0; i < types.length; i++) {
668                                        IType type2 = types[i];
669                                        if (type2.isMember()) {
670                                                tnames.add(type2.getFullyQualifiedName('$'));
671                                        } else {
672                                                tnames.add(type2.getFullyQualifiedName());
673                                        }
674                                }
675                        } catch (JavaModelException e) {
676                                ApiPlugin.log(e.getStatus());
677                        }
678                        updateMonitor(monitor, 0);
679                }
680                // inject removed types inside changed type names so that we can properly detect type removal
681                for (Iterator iterator = this.fAddedRemovedDeltas.iterator(); iterator.hasNext(); ) {
682                        IResourceDelta delta = (IResourceDelta) iterator.next();
683                        if (delta.getKind() != IResourceDelta.REMOVED) continue;
684                        IResource resource = delta.getResource();
685                        IPath typePath = resolveJavaPathFromResource(resource);
686                        if(typePath == null) {
687                                continue;
688                        }
689                        // record removed type names (package + type)
690                        StringBuffer buffer = new StringBuffer();
691                        String[] segments = typePath.segments();
692                        for (int i = 0, max = segments.length; i < max; i++) {
693                                if (i > 0) {
694                                        buffer.append('.');
695                                }
696                                buffer.append(segments[i]);
697                        }
698                        cnames.add(String.valueOf(buffer));
699                }
700                // clean up markers on added deltas
701                if (!this.fAddedRemovedDeltas.isEmpty()) {
702                        IResource manifestFile = Util.getManifestFile(this.fCurrentProject);
703                        if (manifestFile != null) {
704                                try {
705                                        IMarker[] markers = manifestFile.findMarkers(IApiMarkerConstants.COMPATIBILITY_PROBLEM_MARKER, false, IResource.DEPTH_INFINITE);
706                                        for (int i = 0, max = markers.length; i < max; i++) {
707                                                IMarker marker = markers[i];
708                                                String typeName = marker.getAttribute(IApiMarkerConstants.MARKER_ATTR_PROBLEM_TYPE_NAME, null);
709                                                if (typeName != null) {
710                                                        for (Iterator iterator = this.fAddedRemovedDeltas.iterator(); iterator.hasNext(); ) {
711                                                                IResourceDelta delta = (IResourceDelta) iterator.next();
712                                                                if (delta.getKind() != IResourceDelta.ADDED) continue;
713                                                                ICompilationUnit unit = (ICompilationUnit) JavaCore.create(delta.getResource());
714                                                                if(!unit.exists()) {
715                                                                        continue;
716                                                                }
717                                                                IType type = unit.findPrimaryType();
718                                                                if(type == null) {
719                                                                        continue;
720                                                                }
721                                                                if (typeName.equals(type.getFullyQualifiedName())) {
722                                                                        marker.delete();
723                                                                        return;
724                                                                } else {
725                                                                        // check secondary types
726                                                                        try {
727                                                                                types = unit.getAllTypes();
728                                                                                for (int j = 0; j < types.length; j++) {
729                                                                                        IType type2 = types[i];
730                                                                                        String fullyQualifiedName = null;
731                                                                                        if (type2.isMember()) {
732                                                                                                fullyQualifiedName = type2.getFullyQualifiedName('$');
733                                                                                        } else {
734                                                                                                fullyQualifiedName = type2.getFullyQualifiedName();
735                                                                                        }
736                                                                                        if (typeName.equals(fullyQualifiedName)) {
737                                                                                                marker.delete();
738                                                                                                return;
739                                                                                        }
740                                                                                }
741                                                                        } catch (JavaModelException e) {
742                                                                                ApiPlugin.log(e.getStatus());
743                                                                        }
744                                                                }
745                                                        }
746                                                }
747                                        }
748                                } catch (CoreException e) {
749                                        ApiPlugin.log(e.getStatus());
750                                }
751                        }
752                }
753        }

785        /**
786         * Collects the complete set of affected source files from the current project context based on the current JDT build state.
787         * 
788         * @param state
789         */
790        private void collectAffectedSourceFiles(State state) {
791                // the qualifiedStrings are of the form 'p1/p2' & the simpleStrings are just 'X'
792                char[][][] internedQualifiedNames = ReferenceCollection.internQualifiedNames(fPackages);
793                // if a well known qualified name was found then we can skip over these
794                if (internedQualifiedNames.length < fPackages.elementSize) {
795                        internedQualifiedNames = null;
796                }
797                char[][] internedSimpleNames = ReferenceCollection.internSimpleNames(fTypes);
798                // if a well known name was found then we can skip over these
799                if (internedSimpleNames.length < fTypes.elementSize) {
800                        internedSimpleNames = null;
801                }
802                Object[] keyTable = state.getReferences().keyTable;
803                Object[] valueTable = state.getReferences().valueTable;
804                next : for (int i = 0, l = valueTable.length; i < l; i++) {
805                        String typeLocator = (String) keyTable[i];
806                        if (typeLocator != null) {
807                                ReferenceCollection refs = (ReferenceCollection) valueTable[i];
808                                if (refs.includes(internedQualifiedNames, internedSimpleNames)) {
809                                        IFile file = fCurrentProject.getFile(typeLocator);
810                                        if (file == null) {
811                                                continue next;
812                                        }
813                                        if (DEBUG) {
814                                                System.out.println("  adding affected source file " + typeLocator); //$NON-NLS-1$
815                                        }
816                                        fTypesToCheck.add(file);
817                                }
818                        }
819                }
820        }
