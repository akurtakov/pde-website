441         *

576         * and {@link IFieldDescriptor}s.

620                }

640        /**
641         * Returns restriction settings described in the given element.
642         * 
643         * @param project the {@link IJavaProject} context
644         * @param element XML element
645         * @param descriptor the {@link IElementDescriptor} to get the restrictions for
646         * @param earlierversion if the version read from XML is older than the current tooling version
647         * @return restriction settings
648         */
649        private static int getRestrictions(final IJavaProject project, final Element element, final IElementDescriptor descriptor, boolean earlierversion) {
650                int res = RestrictionModifiers.NO_RESTRICTIONS;
651                switch(descriptor.getElementType()) {
652                        case IElementDescriptor.FIELD: {
653                                res = annotateRestriction(element, IApiXmlConstants.ATTR_REFERENCE, RestrictionModifiers.NO_REFERENCE, res);
654                                break;
655                        }
656                        case IElementDescriptor.METHOD: {
657                                IMethodDescriptor method  = (IMethodDescriptor) descriptor;
658                                res = annotateRestriction(element, IApiXmlConstants.ATTR_REFERENCE, RestrictionModifiers.NO_REFERENCE, res);
659                                if(!method.isConstructor()) {
660                                        res = annotateRestriction(element, IApiXmlConstants.ATTR_OVERRIDE, RestrictionModifiers.NO_OVERRIDE, res);
661                                }
662                                break;
663                        }
664                        case IElementDescriptor.TYPE: {
665                                IReferenceTypeDescriptor rtype = (IReferenceTypeDescriptor) descriptor;
666                                res = annotateRestriction(element, IApiXmlConstants.ATTR_IMPLEMENT, RestrictionModifiers.NO_IMPLEMENT, res);
667                                if(earlierversion && RestrictionModifiers.isImplementRestriction(res)) {
668                                        res |= RestrictionModifiers.NO_EXTEND;
669                                }
670                                res = annotateRestriction(element, IApiXmlConstants.ATTR_EXTEND, RestrictionModifiers.NO_EXTEND, res);
671                                if(!RestrictionModifiers.isExtendRestriction(res)) {
672                                        res = annotateRestriction(element, IApiXmlConstants.ATTR_SUBCLASS, RestrictionModifiers.NO_EXTEND, res);
673                                }
674                                res = annotateRestriction(element, IApiXmlConstants.ATTR_INSTANTIATE, RestrictionModifiers.NO_INSTANTIATE, res);
675                                IType type = null;
676                                if (project != null) {
677                                        try {
678                                                type = project.findType(rtype.getQualifiedName());
679                                                if (type != null) {
680                                                        if(Flags.isInterface(type.getFlags())) {
681                                                                res &= ~RestrictionModifiers.NO_INSTANTIATE;
682                                                        }
683                                                        else {
684                                                                res &= ~RestrictionModifiers.NO_IMPLEMENT;
685                                                                if(Flags.isFinal(type.getFlags())) {
686                                                                        res &= ~RestrictionModifiers.NO_EXTEND;
687                                                                }
688                                                                if(Flags.isAbstract(type.getFlags())) {
689                                                                        res &= ~RestrictionModifiers.NO_INSTANTIATE;
690                                                                }
691                                                        }
692                                                }
693                                        } 
694                                        catch (JavaModelException e) {}
695                                }
696                                break;
697                        }
698                }
699                return res;
700        }
