417                }
418                if(DEBUG) {
419                        System.out.println("null filter map, creating a new one"); //$NON-NLS-1$
420                }
421                fFilterMap = new HashMap(5);
422                IPath filepath = getFilterFilePath();
423                IResource file = ResourcesPlugin.getWorkspace().getRoot().findMember(filepath);
424                if(file == null) {
425                        return;
426                }
427                String xml = null;
428                InputStream contents = null;
429                try {
430                        contents = ((IFile)file).getContents();
431                        xml = new String(Util.getInputStreamAsCharArray(contents, -1, IApiCoreConstants.UTF_8));
432                }
433                catch(CoreException ce) {}
434                catch(IOException ioe) {}
435                finally {
436                        if (contents != null) {
437                                try {
438                                        contents.close();
439                                } catch(IOException e) {
440                                        // ignore
441                                }
442                        }
443                }
444                if(xml == null) {
445                        return;
446                }
447                Element root = null;
448                try {
449                        root = Util.parseDocument(xml);
450                }
451                catch(CoreException ce) {
452                        ApiPlugin.log(ce);
453                }
454                if (!root.getNodeName().equals(IApiXmlConstants.ELEMENT_COMPONENT)) {
455                        return;
456                }
457                String component = root.getAttribute(IApiXmlConstants.ATTR_ID);
458                if(component.length() == 0) {
459                        return;
460                }
461                String versionValue = root.getAttribute(IApiXmlConstants.ATTR_VERSION);
462                int currentVersion = Integer.parseInt(IApiXmlConstants.API_FILTER_STORE_CURRENT_VERSION);
463                int version = 0;
464                if(versionValue.length() != 0) {
465                        try {
466                                version = Integer.parseInt(versionValue);
467                        } catch (NumberFormatException e) {
468                                // ignore
469                        }
470                }
471                if (version != currentVersion) {
472                        // we discard all filters since there is no way to retrieve the type name
473                        fNeedsSaving = true;
474                        persistApiFilters();
475                        return;
476                }
477                NodeList resources = root.getElementsByTagName(IApiXmlConstants.ELEMENT_RESOURCE);
478                ArrayList newfilters = new ArrayList();
479                for(int i = 0; i < resources.getLength(); i++) {
480                        Element element = (Element) resources.item(i);
481                        String path = element.getAttribute(IApiXmlConstants.ATTR_PATH);
482                        if(path.length() == 0) {
483                                continue;
484                        }
485                        String typeName = element.getAttribute(IApiXmlConstants.ATTR_TYPE);
486                        if (typeName.length() == 0) {
487                                typeName = null;
488                        }
489                        IProject project = (IProject) ResourcesPlugin.getWorkspace().getRoot().findMember(component);
490                        if(project == null) {
491                                continue;
492                        }
493                        IResource resource = project.findMember(new Path(path));
494                        if(resource == null) {
495                                continue;
496                        }
497                        NodeList filters = element.getElementsByTagName(IApiXmlConstants.ELEMENT_FILTER);
498                        for(int j = 0; j < filters.getLength(); j++) {
499                                element = (Element) filters.item(j);
500                                int id = loadIntegerAttribute(element, IApiXmlConstants.ATTR_ID);
501                                if(id <= 0) {
502                                        continue;
503                                }
504                                String[] messageargs = null;
505                                NodeList elements = element.getElementsByTagName(IApiXmlConstants.ELEMENT_PROBLEM_MESSAGE_ARGUMENTS);
506                                if (elements.getLength() != 1) continue;
507                                Element messageArguments = (Element) elements.item(0);
508                                NodeList arguments = messageArguments.getElementsByTagName(IApiXmlConstants.ELEMENT_PROBLEM_MESSAGE_ARGUMENT);
509                                int length = arguments.getLength();
510                                messageargs = new String[length];
511                                for (int k = 0; k < length; k++) {
512                                        Element messageArgument = (Element) arguments.item(k);
513                                        messageargs[k] = messageArgument.getAttribute(IApiXmlConstants.ATTR_VALUE);
514                                }
515                                newfilters.add(ApiProblemFactory.newApiProblem(resource.getProjectRelativePath().toPortableString(),
516                                                typeName,
517                                                messageargs, null, null, -1, -1, -1, id));
518                        }
519                }
520                internalAddFilters((IApiProblem[]) newfilters.toArray(new IApiProblem[newfilters.size()]), false);
521                newfilters.clear();
522        }
523        
524        /**
525         * Internal use method that allows auto-persisting of the filter file to be turned on or off
526         * @param problems the problems to add the the store
527         * @param persist if the filters should be auto-persisted after they are added
528         */
529        private void internalAddFilters(IApiProblem[] problems, boolean persist) {
530                if(problems == null) {
531                        if(DEBUG) {
532                                System.out.println("null problems array not addding filters"); //$NON-NLS-1$
533                        }
534                        return;
535                }
536                initializeApiFilters();
537                Set filters = null;
538                for(int i = 0; i < problems.length; i++) {
539                        IApiProblem problem = problems[i];
540                        IApiProblemFilter filter = new ApiProblemFilter(fProject.getElementName(), problem);
541                        String resourcePath = problem.getResourcePath();
542                        if (resourcePath == null) {
543                                continue;
544                        }
545                        IResource resource = fProject.getProject().findMember(new Path(resourcePath));
546                        if(resource == null) {
547                                continue;
548                        }
549                        Map pTypeNames = (Map) fFilterMap.get(resource);
550                        String typeName = problem.getTypeName();
551                        if (typeName == null) {
552                                typeName = GLOBAL;
553                        }
554                        if(pTypeNames == null) {
555                                filters = new HashSet();
556                                pTypeNames = new HashMap();
557                                pTypeNames.put(typeName, filters);
558                                fFilterMap.put(resource, pTypeNames);
559                        } else {
560                                filters = (Set) pTypeNames.get(typeName);
561                                if (filters == null) {
562                                        filters = new HashSet();
563                                        pTypeNames.put(typeName, filters);
564                                }
565                        }
566                        fNeedsSaving |= filters.add(filter);
567                }
568                if(persist) {
569                        persistApiFilters();
570                }
571        }

573        /**
574         * Loads the specified integer attribute from the given xml element
575         * @param element
576         * @param name
577         * @return
578         */
579        private static int loadIntegerAttribute(Element element, String name) {
580                String value = element.getAttribute(name);
581                if(value.length() == 0) {
582                        return -1;
583                }
584                try {
585                        int number = Integer.parseInt(value);
586                        return number;
587                }
588                catch(NumberFormatException nfe) {}
589                return -1;
590        }

610         */
611        public void resourceChanged(IResourceChangeEvent event) {
612                if(fTriggeredChange) {
613                        //eat the event if the deletion / addition / change occurred because we persisted the file
614                        //via the persistApiFilters(..) method
615                        //see https://bugs.eclipse.org/bugs/show_bug.cgi?id=222442
616                        fTriggeredChange = false;
617                        return;
618                }
619                if(event.getType() == IResourceChangeEvent.POST_CHANGE) {
620                        if(event.getType() == IResourceChangeEvent.POST_CHANGE) {
621                                IPath path = getFilterFilePath();
622                                IResourceDelta leafdelta = event.getDelta().findMember(path);
623                                if(leafdelta == null) {
624                                        return;
625                                }
626                                boolean needsbuild = false;
627                                if(leafdelta.getKind() == IResourceDelta.REMOVED) {
628                                        if(fFilterMap != null) {
629                                                fFilterMap.clear();
630                                                needsbuild = true;
631                                        }
632                                }
633                                else if(leafdelta.getKind() == IResourceDelta.ADDED || 
634                                                (leafdelta.getFlags() & IResourceDelta.CONTENT) != 0 || 
635                                                (leafdelta.getFlags() & IResourceDelta.REPLACED) != 0) {
636                                        IResource resource = leafdelta.getResource();
637                                        if(resource != null && resource.getType() == IResource.FILE) {
638                                                IFile file = (IFile) resource;
639                                                if(file.isAccessible()) {
640                                                        try {
641                                                                if(fFilterMap != null) {
642                                                                        fFilterMap.clear();
643                                                                        fFilterMap = null; 
644                                                                }
645                                                                initializeApiFilters();
646                                                        }
647                                                        finally {
648                                                                needsbuild = true;
649                                                        }
650                                                }
651                                        }
652                                }
653                                if(needsbuild && ResourcesPlugin.getWorkspace().isAutoBuilding()) {
654                                        Util.getBuildJob(new IProject[] {fProject.getProject()}).schedule();
655                                }
656                        }
657                }
658        }
