427        public boolean considerReference(IReference reference) {
428                IApiComponent apiComponent = reference.getMember().getApiComponent();

432                // extract the package name

435                try {

437                        IApiComponent[] resolvePackages = baseline.resolvePackage(apiComponent, substring);
438                        switch(resolvePackages.length) {
439                                case 1 :
