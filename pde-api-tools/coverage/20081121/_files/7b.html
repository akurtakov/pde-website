466         */
467        private static boolean isInterfaceCompatible(IDelta delta) {
468                switch(delta.getKind()) {
469                        case IDelta.ADDED :
470                                switch(delta.getFlags()) {
471                                        case IDelta.FIELD :
472                                                return RestrictionModifiers.isImplementRestriction(delta.getRestrictions()) &&
473                                                           RestrictionModifiers.isExtendRestriction(delta.getRestrictions());
474                                        case IDelta.METHOD :
475                                        case IDelta.SUPER_INTERFACE_WITH_METHODS :
476                                                return RestrictionModifiers.isImplementRestriction(delta.getRestrictions());
477                                        case IDelta.TYPE_PARAMETER :
478                                                return false;
479                                        case IDelta.RESTRICTIONS :
480                                                return false;
481                                }
482                                break;
483                        case IDelta.REMOVED :
484                                switch(delta.getFlags()) {
485                                        case IDelta.FIELD :
486                                        case IDelta.METHOD :
487                                        case IDelta.API_FIELD :
488                                        case IDelta.API_METHOD :
489                                        case IDelta.TYPE_MEMBER :
490                                        case IDelta.TYPE_PARAMETER :
491                                                return false;
492                                }
493                                break;
494                        case IDelta.CHANGED :
495                                switch(delta.getFlags()) {
496                                        case IDelta.CONTRACTED_SUPERINTERFACES_SET :
497                                        case IDelta.TYPE_CONVERSION :
498                                                return false;
499                                }
500                                break;
501                }
502                return true;
503        }
