641        /**
642         * Returns the complete listing of required projects from the classpath of the backing project
643         * @param includeBinaryPrerequisites
644         * @return
645         * @throws CoreException
646         */
647        private IProject[] getRequiredProjects(boolean includebinaries) throws CoreException {
648                IWorkspaceRoot workspaceRoot = ResourcesPlugin.getWorkspace().getRoot();
649                if (this.currentproject == null || workspaceRoot == null) { 
650                        return new IProject[0];
651                }
652                ArrayList projects = new ArrayList();
653                try {
654                        IJavaProject javaProject = JavaCore.create(this.currentproject);
655                        HashSet blocations = new HashSet();
656                        blocations.add(javaProject.getOutputLocation());
657                        projecttooutputlocations.put(this.currentproject, blocations);
658                        IClasspathEntry[] entries = javaProject.getResolvedClasspath(true);
659                        for (int i = 0, l = entries.length; i < l; i++) {
660                                IClasspathEntry entry = entries[i];
661                                IPath path = entry.getPath();
662                                IProject p = null;
663                                switch (entry.getEntryKind()) {
664                                        case IClasspathEntry.CPE_PROJECT : {
665                                                p = workspaceRoot.getProject(path.lastSegment()); // missing projects are considered too
666                                                if (isOptional(entry) && !p.hasNature(ApiPlugin.NATURE_ID)) {// except if entry is optional 
667                                                        p = null;
668                                                }
669                                                break;
670                                        }
671                                        case IClasspathEntry.CPE_LIBRARY : {
672                                                if (includebinaries && path.segmentCount() > 1) {
673                                                        // some binary resources on the class path can come from projects that are not included in the project references
674                                                        IResource resource = workspaceRoot.findMember(path.segment(0));
675                                                        if (resource instanceof IProject) {
676                                                                p = (IProject) resource;
677                                                        }
678                                                }
679                                                break;
680                                        }
681                                        case IClasspathEntry.CPE_SOURCE: {
682                                                IPath entrypath = entry.getOutputLocation();
683                                                if(entrypath != null) {
684                                                        blocations.add(entrypath);
685                                                }
686                                        }
687                                }
688                                if (p != null && !projects.contains(p)) {
689                                        projects.add(p);
690                                        //try to derive all of the output locations for each of the projects
691                                        javaProject = JavaCore.create(p);
692                                        HashSet bins = new HashSet();
693                                        if(javaProject.exists()) {
694                                                bins.add(javaProject.getOutputLocation());
695                                                IClasspathEntry[] source = javaProject.getRawClasspath();
696                                                IPath entrypath = null;
697                                                for(int j = 0; j < source.length; j++) {
698                                                        if(source[j].getEntryKind() == IClasspathEntry.CPE_SOURCE) {
699                                                                entrypath = source[j].getOutputLocation();
700                                                                if(entrypath != null) {
701                                                                        bins.add(entrypath);
702                                                                }
703                                                        }
704                                                }
705                                                this.projecttooutputlocations.put(p, bins);
706                                        }
707                                }
708                        }
709                } 
710                catch(JavaModelException e) {
711                        return new IProject[0];
712                }
713                IProject[] result = new IProject[projects.size()];
714                projects.toArray(result);
715                return result;
716        }
