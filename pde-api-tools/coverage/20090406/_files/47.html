392                                        List fragments = new ArrayList(1);
393                                        for (int i = 0; i < roots.length; i++) {
394                                                IPackageFragmentRoot root = roots[i];
395                                                IClasspathEntry entry = root.getRawClasspathEntry();
396                                                switch (entry.getEntryKind()) {
397                                                case IClasspathEntry.CPE_SOURCE:
398                                                case IClasspathEntry.CPE_LIBRARY:
399                                                        IPackageFragment fragment = root.getPackageFragment(pkg.getName());
400                                                        if (fragment.exists()) {
401                                                                fragments.add(fragment);
402                                                        }
403                                                        break;
404                                                default:
405                                                        if (!root.isArchive()
406                                                                        && root.getKind() == IPackageFragmentRoot.K_BINARY) {
407                                                                // class file folder
408                                                                fragment = root.getPackageFragment(pkg.getName());
409                                                                if (fragment.exists()) {
410                                                                        fragments.add(fragment);
411                                                                }
412                                                        }
413                                                }
414                                        }
415                                        if (fragments.isEmpty()) {
416                                                return null;
417                                        } else {
418                                                return newPackageNode((IPackageFragment[])fragments.toArray(new IPackageFragment[fragments.size()]), parentNode, element, VisibilityModifiers.PRIVATE, RestrictionModifiers.NO_RESTRICTIONS);
419                                        }
420                                        
421                                } catch (CoreException e) {
422                                        return null;
423                                }
424                        case IElementDescriptor.TYPE:
425                                IReferenceTypeDescriptor descriptor = (IReferenceTypeDescriptor) element;
426                                try {
427                                        IType type = null;
428                                        String name = descriptor.getName();
429                                        if (parentNode instanceof PackageNode) {
430                                                IPackageFragment[] fragments = ((PackageNode) parentNode).fFragments; 
431                                                for (int i = 0; i < fragments.length; i++) {
432                                                        IPackageFragment fragment = fragments[i];
433                                                        if (fragment.getKind() == IPackageFragmentRoot.K_SOURCE) {
434                                                                ICompilationUnit unit = fragment.getCompilationUnit(name + ".java"); //$NON-NLS-1$
435                                                                try {
436                                                                        IResource resource = unit.getUnderlyingResource();
437                                                                        if (resource != null) {
438                                                                                type = unit.getType(name);
439                                                                        }
440                                                                } catch (JavaModelException jme) {
441                                                                        // exception if the resource does not exist
442                                                                        if (!jme.getJavaModelStatus().isDoesNotExist()) {
443                                                                                throw jme;
444                                                                        }
445                                                                }
446                                                        } else {
447                                                                IClassFile file = fragment.getClassFile(name + ".class"); //$NON-NLS-1$
448                                                                if (file.exists()) {
449                                                                        type = file.getType();
450                                                                }
451                                                        }
452                                                }
453                                        } else if (parentNode instanceof TypeNode) {
454                                                type = ((TypeNode)parentNode).fType.getType(name);
455                                        }
456                                        if (type != null) {
457                                                return newTypeNode(type, parentNode, element, VISIBILITY_INHERITED, RestrictionModifiers.NO_RESTRICTIONS);
458                                        }
459                                } catch (CoreException e ) {
460                                        return null;
461                                }
462                                return null;
463                }
464                return super.createNode(parentNode, element);
465        }

508        /**
509         * Refreshes package nodes if required.
510         */
511        private synchronized void refreshPackages() {
512                if (fRefreshingInProgress) {
513                        return;
514                }
515                // check if in synch
516                if (fManifestFile == null || (fManifestFile.getModificationStamp() != fPackageTimeStamp)) {
517                        try {
518                                modified();
519                                fRefreshingInProgress = true;
520                                // set all existing packages to PRIVATE (could clear
521                                // the map, but it would be less efficient)
522                                Iterator iterator = fPackageMap.values().iterator();
523                                while (iterator.hasNext()) {
524                                        PackageNode node = (PackageNode) iterator.next();
525                                        node.visibility = VisibilityModifiers.PRIVATE;
526                                }
527                                fManifestFile = getJavaProject().getProject().getFile(JarFile.MANIFEST_NAME);
528                                if (fManifestFile.exists()) {
529                                        try {
530                                                IPackageFragment[] fragments = getLocalPackageFragments();
531                                                Set names = new HashSet();
532                                                for (int i = 0; i < fragments.length; i++) {
533                                                        names.add(fragments[i].getElementName());
534                                                }
535                                                BundleApiComponent.initializeApiDescription(this, fBundle, names);
536                                                fPackageTimeStamp = fManifestFile.getModificationStamp();
537                                        } catch (CoreException e) {
538                                                ApiPlugin.log(e.getStatus());
539                                        }
540                                }
541                        } finally {
542                                fRefreshingInProgress = false;
543                        }
544                }
545        }
