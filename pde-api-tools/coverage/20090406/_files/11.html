434                Element kelement = null;
435                Integer kind = new Integer(reference.getReferenceKind());
436                kelement = findKindElement(parent, kind);

438                        kelement = document.createElement(IApiXmlConstants.REFERENCE_KIND);
439                        kelement.setAttribute(IApiXmlConstants.ATTR_REFERENCE_KIND_NAME, Reference.getReferenceText(kind.intValue()));
440                        kelement.setAttribute(IApiXmlConstants.ATTR_KIND, kind.toString());
441                        parent.appendChild(kelement);
442                }
443                Element relement = document.createElement(IApiXmlConstants.ATTR_REFERENCE);
444                IApiMember member = reference.getMember();
445                relement.setAttribute(IApiXmlConstants.ATTR_ORIGIN, getText(member));
446                member = reference.getResolvedReference();
447                if(member != null) {
448                        relement.setAttribute(IApiXmlConstants.ATTR_REFEREE, getText(member));
449                        relement.setAttribute(IApiXmlConstants.ATTR_LINE_NUMBER, Integer.toString(reference.getLineNumber()));
450                        String sig = reference.getReferencedSignature();
451                        if(sig != null) {
452                                relement.setAttribute(IApiXmlConstants.ATTR_SIGNATURE, sig);
453                        }
454                        kelement.appendChild(relement);
455                }
456        }

468                        case IApiElement.FIELD: return Signatures.getQualifiedFieldSignature((IApiField) member);
