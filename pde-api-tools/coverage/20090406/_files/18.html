462         * @param signature2
463         * @return true if the signatures are equal, false otherwise
464         */
465        public static boolean matchesSignatures(String signature, String signature2) {
466                if (!matches(Signature.getReturnType(signature), Signature.getReturnType(signature2))) {
467                        return false;
468                }
469                String[] parameterTypes = Signature.getParameterTypes(signature);
470                String[] parameterTypes2 = Signature.getParameterTypes(signature2);
471                int length = parameterTypes.length;
472                int length2 = parameterTypes2.length;
473                if (length != length2) return false;
474                for (int i = 0; i < length2; i++) {
475                        if (!matches(parameterTypes[i], parameterTypes2[i])) {
476                                return false;
477                        }
478                }
479                return true;
480        }

482        /**
483         * Returns if the two types match. Types are considered to match 
484         * iff the type name and array count (if any) are the same
485         * @param type
486         * @param type2
487         * @return true if the type names match, false otherwise
488         */
489        private static boolean matches(String type, String type2) {
490                if (Signature.getArrayCount(type) == Signature.getArrayCount(type2)) {
491                        String el1 = Signature.getElementType(type);
492                        String el2 = Signature.getElementType(type2);
493                        String[] typeargs1 = Signature.getTypeArguments(el1);
494                        String[] typeargs2 = Signature.getTypeArguments(el2);
495                        if(typeargs1.length == typeargs2.length) {
496                                if(typeargs1.length > 0) {
497                                        for(int i = 0; i < typeargs1.length; i++) {
498                                                if(!matches(typeargs1[i], typeargs2[i])) {
499                                                        return false;
500                                                }
501                                        }
502                                        return true;
503                                }
504                                else {
505                                        String signatureSimpleName = Signature.getSignatureSimpleName(el1);
506                                        String signatureSimpleName2 = Signature.getSignatureSimpleName(el2);
507                                        if (signatureSimpleName.equals(signatureSimpleName2)) {
508                                                return true;
509                                        }
510                                        int index = signatureSimpleName2.lastIndexOf('.');
511                                        if (index != -1) {
512                                                // the right side is a member type
513                                                return signatureSimpleName.equals(signatureSimpleName2.subSequence(index + 1, signatureSimpleName2.length()));
514                                        }
515                                }
516                        }
517                }
518                return false;
519        }

545        /**
546         * Collects the synthetic parameter of the fully qualified name of the enclosing context for a constructor of an inner type 
547         * @param method the constructor declaration
548         * @param rparams the listing of parameters to add to
549         */
550        static void collectSyntheticParam(final MethodDeclaration method, List rparams) {
551                Assert.isNotNull(method);
552                if(Signatures.isInTopLevelType(method)) {
553                        return;
554                }
555                ASTNode parent = method.getParent();
556                AbstractTypeDeclaration type = (AbstractTypeDeclaration) parent;
557                if (Signatures.isStatic(type)) {
558                        // if the type is static it doesn't need the enclosing type
559                        return;
560                }
561                StringBuffer name = new StringBuffer();
562                while(parent != null) {
563                        parent = parent.getParent();
564                        if(parent instanceof AbstractTypeDeclaration) {
565                                type = (AbstractTypeDeclaration) parent;
566                                name.insert(0, type.getName().getFullyQualifiedName());
567                                if(type.isMemberTypeDeclaration()) {
568                                        name.insert(0, '$');
569                                }
570                                continue;
571                        }
572                        if(parent instanceof CompilationUnit) {
573                                CompilationUnit cunit = (CompilationUnit) parent;
574                                PackageDeclaration pdec = cunit.getPackage();
575                                if(pdec != null) {
576                                        name.insert(0, '.');
577                                        name.insert(0, cunit.getPackage().getName().getFullyQualifiedName());
578                                }
579                        }
580                }
581                name.insert(0, "L"); //$NON-NLS-1$
582                name.append(';');
583                if(name.length() > 2) {
584                        rparams.add(0, name.toString());
585                }
586        }

589         * Returns if the {@link AbstractTypeDeclaration} is static or not (has the static
590         * keyword or not)
591         * 
592         * @param typeDeclaration
593         * @return true if it is static, false otherwise
594         */
595        static boolean isStatic(AbstractTypeDeclaration typeDeclaration) {
596                List modifiers = typeDeclaration.modifiers();
597                if (modifiers.isEmpty()) return false;
598                for (Iterator iterator = modifiers.iterator(); iterator.hasNext(); ) {
599                        IExtendedModifier modifier = (IExtendedModifier) iterator.next();
600                        if (!modifier.isModifier()) {
601                                continue;
602                        }
603                        Modifier modifier2 = (Modifier) modifier;
604                        if (modifier2.isStatic()) return true;
605                }
606                return false;
607        }
