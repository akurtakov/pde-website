402                                                if (pool == null) {
403                                                        pool = new ArrayList();
404                                                        pools.put(dir, pool);
405                                                }
406                                                pool.add(components[i]);
407                                        } else {
408                                                unRooted.add(components[i]);
409                                        }
410                                }
411                                
412                        }
413                }
414                Document document = Util.newDocument();
415                Element root = document.createElement(IApiXmlConstants.ELEMENT_APIPROFILE);
416                document.appendChild(root);
417                root.setAttribute(IApiXmlConstants.ATTR_NAME, profile.getName());
418                root.setAttribute(IApiXmlConstants.ATTR_VERSION, IApiXmlConstants.API_PROFILE_CURRENT_VERSION);
419                // dump component pools
420                Element subroot = null;
421                File dir = null;
422                List comps = null;
423                IApiComponent comp = null;
424                Element celement = null;
425                for(Iterator iter = pools.entrySet().iterator(); iter.hasNext();) {
426                        Map.Entry entry = (Map.Entry) iter.next();
427                        dir = (File) entry.getKey();
428                        subroot = document.createElement(IApiXmlConstants.ELEMENT_POOL);
429                        root.appendChild(subroot);
430                        subroot.setAttribute(IApiXmlConstants.ATTR_LOCATION, new Path(dir.getAbsolutePath()).toPortableString());
431                        comps = (List) entry.getValue();
432                        for(Iterator iter2 = comps.iterator(); iter2.hasNext();) {
433                                comp = (IApiComponent) iter2.next();
434                                celement = document.createElement(IApiXmlConstants.ELEMENT_APICOMPONENT);
435                                celement.setAttribute(IApiXmlConstants.ATTR_ID, comp.getId());
436                                celement.setAttribute(IApiXmlConstants.ATTR_VERSION, comp.getVersion());
437                                subroot.appendChild(celement);
438                        }
439                }
440                // dump un-pooled components
441                for(Iterator iter = unRooted.iterator(); iter.hasNext();) {
442                        comp = (IApiComponent) iter.next();
443                        celement = document.createElement(IApiXmlConstants.ELEMENT_APICOMPONENT);
444                        celement.setAttribute(IApiXmlConstants.ATTR_ID, comp.getId());
445                        celement.setAttribute(IApiXmlConstants.ATTR_VERSION, comp.getVersion());
446                        celement.setAttribute(IApiXmlConstants.ATTR_LOCATION, new Path(comp.getLocation()).toPortableString());
447                        root.appendChild(celement);
448                }
449                return Util.serializeDocument(document);
450        }

466         * 
467         * @param stream input stream
468         * @return API profile
469         * @throws CoreException if unable to restore the profile
470         */
471        private void restoreBaseline(IApiBaseline baseline, InputStream stream) throws CoreException {
472                long start = System.currentTimeMillis();
473                DocumentBuilder parser = null;
474                try {
475                        parser = DocumentBuilderFactory.newInstance().newDocumentBuilder();
476                        parser.setErrorHandler(new DefaultHandler());
477                } catch (ParserConfigurationException e) {
478                        abort("Error restoring API baseline", e); //$NON-NLS-1$
479                } catch (FactoryConfigurationError e) {
480                        abort("Error restoring API baseline", e); //$NON-NLS-1$
481                }
482                try {
483                        Document document = parser.parse(stream);
484                        Element root = document.getDocumentElement();
485                        if(root.getNodeName().equals(IApiXmlConstants.ELEMENT_APIPROFILE)) {
486                                // un-pooled components
487                                NodeList children = root.getElementsByTagName(IApiXmlConstants.ELEMENT_APICOMPONENT);
488                                List components = new ArrayList();
489                                for(int j = 0; j < children.getLength(); j++) {
490                                        Element componentNode = (Element) children.item(j);
491                                        // this also contains components in pools, so don't process them
492                                        if (componentNode.getParentNode().equals(root)) {
493                                                String location = componentNode.getAttribute(IApiXmlConstants.ATTR_LOCATION);
494                                                IApiComponent component = ApiModelFactory.newApiComponent(baseline, Path.fromPortableString(location).toOSString());
495                                                if(component != null) {
496                                                        components.add(component);
497                                                }
498                                        }
499                                }
500                                // pooled components
501                                children = root.getElementsByTagName(IApiXmlConstants.ELEMENT_POOL);
502                                IApiComponent component = null;
503                                for(int j = 0; j < children.getLength(); j++) {
504                                        String location = ((Element) children.item(j)).getAttribute(IApiXmlConstants.ATTR_LOCATION);
505                                        IPath poolPath = Path.fromPortableString(location);
506                                        NodeList componentNodes = root.getElementsByTagName(IApiXmlConstants.ELEMENT_APICOMPONENT);
507                                        for (int i = 0; i < componentNodes.getLength(); i++) {
508                                                Element compElement = (Element) componentNodes.item(i);
509                                                String id = compElement.getAttribute(IApiXmlConstants.ATTR_ID);
510                                                String ver = compElement.getAttribute(IApiXmlConstants.ATTR_VERSION);
511                                                StringBuffer name = new StringBuffer();
512                                                name.append(id);
513                                                name.append('_');
514                                                name.append(ver);
515                                                File file = poolPath.append(name.toString()).toFile();
516                                                if (!file.exists()) {
517                                                        name.append(".jar"); //$NON-NLS-1$
518                                                        file = poolPath.append(name.toString()).toFile();
519                                                }
520                                                component = ApiModelFactory.newApiComponent(baseline, file.getAbsolutePath());
521                                                if(component != null) {
522                                                        components.add(component);
523                                                }
524                                        }
525                                }
526                                baseline.addApiComponents((IApiComponent[]) components.toArray(new IApiComponent[components.size()]));
527                        }
528                } catch (IOException e) {
529                        abort("Error restoring API baseline", e); //$NON-NLS-1$
530                } catch(SAXException e) {
531                        abort("Error restoring API baseline", e); //$NON-NLS-1$
532                } finally {
533                        try {
534                                stream.close();
535                        } catch (IOException io) {
536                                ApiPlugin.log(io);
537                        }
538                }
539                if (baseline == null) {
540                        abort("Invalid profile description", null); //$NON-NLS-1$
541                }
542                if(DEBUG) {
543                        System.out.println("Time to restore a persisted profile : " + (System.currentTimeMillis() - start) + "ms"); //$NON-NLS-1$ //$NON-NLS-2$
544                }
545        }

733        /**
734         * Processes the java element deltas of interest
735         * @param deltas
736         */
737        private synchronized void processJavaElementDeltas(IJavaElementDelta[] deltas, IJavaProject project) {
738                try {
739                        IJavaElementDelta delta = null;
740                        for(int i = 0; i < deltas.length; i++) {
741                                delta = deltas[i];
742                                switch(delta.getElement().getElementType()) {
743                                        case IJavaElement.JAVA_PROJECT: {
744                                                IJavaProject proj = (IJavaProject) delta.getElement();
745                                                IProject pj = proj.getProject();
746                                                if (acceptProject(pj)) {
747                                                        int flags = delta.getFlags();
748                                                        switch (delta.getKind()) {
749                                                                //process the project changed only if the project is API aware
750                                                                case IJavaElementDelta.CHANGED: {
751                                                                        if( (flags & IJavaElementDelta.F_RESOLVED_CLASSPATH_CHANGED) != 0 ||
752                                                                                (flags & IJavaElementDelta.F_CLASSPATH_CHANGED) != 0 ||
753                                                                                (flags & IJavaElementDelta.F_CLOSED) != 0 ||
754                                                                                (flags & IJavaElementDelta.F_OPENED) != 0) {
755                                                                                        if(DEBUG) {
756                                                                                                System.out.println("--> processing CLASSPATH CHANGE/CLOSE/OPEN project: ["+proj.getElementName()+"]"); //$NON-NLS-1$ //$NON-NLS-2$
757                                                                                        }
758                                                                                        disposeWorkspaceBaseline();
759                                                                        } else if((flags & IJavaElementDelta.F_CHILDREN) != 0) {
760                                                                                if(DEBUG) {
761                                                                                        System.out.println("--> processing child deltas of project: ["+proj.getElementName()+"]"); //$NON-NLS-1$ //$NON-NLS-2$
762                                                                                }
763                                                                                processJavaElementDeltas(delta.getAffectedChildren(), proj);
764                                                                        }
765                                                                        break;
766                                                                }
767                                                                case IJavaElementDelta.REMOVED: {
768                                                                        if((flags & IJavaElementDelta.F_MOVED_TO) != 0) {
769                                                                                if(DEBUG) {
770                                                                                        System.out.println("--> processing PROJECT RENAME: ["+proj.getElementName()+"]"); //$NON-NLS-1$ //$NON-NLS-2$
771                                                                                }
772                                                                                disposeWorkspaceBaseline();
773                                                                        }
774                                                                        break;
775                                                                }
776                                                        }
777                                                }
778                                                break;
779                                        }
780                                        case IJavaElement.PACKAGE_FRAGMENT_ROOT: {
781                                                IPackageFragmentRoot root = (IPackageFragmentRoot) delta.getElement();
782                                                if(DEBUG) {
783                                                        System.out.println("processed package fragment root delta: ["+root.getElementName()+"]"); //$NON-NLS-1$ //$NON-NLS-2$
784                                                }
785                                                switch(delta.getKind()) {
786                                                        case IJavaElementDelta.CHANGED: {
787                                                                if(DEBUG) {
788                                                                        System.out.println("processed children of CHANGED package fragment root: ["+root.getElementName()+"]"); //$NON-NLS-1$ //$NON-NLS-2$
789                                                                }
790                                                                processJavaElementDeltas(delta.getAffectedChildren(), project);
791                                                                break;
792                                                        }
793                                                }
794                                                break;
795                                        }
796                                        case IJavaElement.PACKAGE_FRAGMENT: {
797                                                IPackageFragment fragment = (IPackageFragment) delta.getElement();
798                                                if(delta.getKind() == IJavaElementDelta.REMOVED) {
799                                                        handlePackageRemoval(project.getProject(), fragment);
800                                                }
801                                                break;
802                                        }
803                                }
804                        }
805                } catch (CoreException e) {
806                        ApiPlugin.log(e);
807                }
808        }

811         * Handles the specified {@link IPackageFragment} being removed.
812         * When a packaged is removed, we:

869                if (delta != null) {
870                        IResourceDelta[] children = delta.getAffectedChildren(IResourceDelta.CHANGED);
871                        for (int i = 0; i < children.length; i++) {
872                                IResourceDelta d = children[i];
873                                IResource resource = d.getResource();
874                                if (resource.getType() == IResource.PROJECT) {
875                                        if ((d.getFlags() & IResourceDelta.DESCRIPTION) != 0) {
876                                                IProject project = (IProject)resource;
877                                                if (project.isAccessible()) {
878                                                        try {
879                                                                if (!project.getDescription().hasNature(ApiPlugin.NATURE_ID)) {
880                                                                        IJavaProject jp = JavaCore.create(project);
881                                                                        if (jp.exists()) {
882                                                                                ApiDescriptionManager.getDefault().clean(jp, true, true);
883                                                                                ApiAnalysisBuilder.cleanupMarkers(resource);
884                                                                        }
885                                                                }
886                                                        } catch (CoreException e) {
887                                                                ApiPlugin.log(e.getStatus());
888                                                        }
889                                                }

891                                        }
892                                }
893                        }
