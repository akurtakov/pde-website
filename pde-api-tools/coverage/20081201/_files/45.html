428                        IApiComponent apiComponent = member.getApiComponent();
429                        String[] lowestEEs = apiComponent.getLowestEEs();
430                        if (lowestEEs == null) {
431                                // this should not be true for Eclipse bundle as they should always have a EE set
432                                return false;
433                        }
434                        loop: for (int i = 0, max = lowestEEs.length; i < max; i++) {
435                                String lowestEE = lowestEEs[i];
436                                int eeValue = ProfileModifiers.getValue(lowestEE); 
437                                if (eeValue == ProfileModifiers.NO_PROFILE_VALUE) {
438                                        return false;
439                                }
440                                IApiMember resolvedReference = reference.getResolvedReference();
441                                IElementDescriptor elementDescriptor = resolvedReference.getHandle();
442                                IApiDescription systemApiDescription = apiComponent.getSystemApiDescription(eeValue);
443                                boolean value = !Util.isAPI(eeValue, elementDescriptor, systemApiDescription);
444                                if (value) {
445                                        /*
446                                         * Make sure that the resolved reference doesn't below to one of the imported package of
447                                         * the current component
448                                         */
449                                        if (apiComponent instanceof BundleApiComponent) {
450                                                BundleDescription bundle = ((BundleApiComponent)apiComponent).getBundleDescription();
451                                                ImportPackageSpecification[] importPackages = bundle.getImportPackages();
452                                                String packageName = getPackageName(elementDescriptor);
453                                                for (int j = 0, max2 = importPackages.length; j < max2; j++) {
454                                                        ImportPackageSpecification importPackageSpecification = importPackages[j];
455                                                        // get the IPackageDescriptor for the element descriptor
456                                                        String importPackageName = importPackageSpecification.getName();
457                                                        if (importPackageName.equals(packageName)) {
458                                                                continue loop;
459                                                        }
460                                                }
461                                        }
462                                        if (this.referenceEEs == null) {
463                                                this.referenceEEs = new HashMap(3);
464                                        }
465                                        this.referenceEEs.put(reference, new Integer(eeValue));
466                                        return true;
467                                }
468                        }
469                } catch (CoreException e) {
470                        ApiPlugin.log(e);
471                }
472                return false;
473        }
474        private static String getPackageName(IElementDescriptor elementDescriptor) {

476                while (currentDescriptor != null && currentDescriptor.getElementType() != IElementDescriptor.PACKAGE) {

478                }

481        public boolean considerReference(IReference reference) {
482                try {
483                        IApiComponent apiComponent = reference.getMember().getApiComponent();
484                        IApiBaseline baseline = apiComponent.getBaseline();
485                        if (baseline == null) return false;
486                        String referencedTypeName = reference.getReferencedTypeName();
487                        // extract the package name
488                        int index = referencedTypeName.lastIndexOf('.');
489                        if (index == -1) return false;
490                        String substring = referencedTypeName.substring(0, index);
491                        IApiComponent[] resolvePackages = baseline.resolvePackage(apiComponent, substring);
492                        switch(resolvePackages.length) {
493                                case 1 :
494                                        if (resolvePackages[0].isSystemComponent()) {
495                                                switch(reference.getReferenceKind()) {
496                                                        case ReferenceModifiers.REF_OVERRIDE :
497                                                        case ReferenceModifiers.REF_CONSTANTPOOL :
498                                                                return false;
499                                                }
500                                                retainReference(reference);
501                                                return true;
502                                        }
503                        }
504                } catch (CoreException e) {
505                        ApiPlugin.log(e);
506                }
507                return false;
508        }
