415                /**
416                 * Returns a method descriptor with a resolved signature for the given method
417                 * descriptor with an unresolved signature.
418                 * 
419                 * @param descriptor method to resolve
420                 * @return resolved method descriptor or the same method descriptor if unable to
421                 *         resolve
422                 * @exception CoreException if unable to resolve the method and a class file
423                 *  container was provided for this purpose
424                 */
425                private IMethodDescriptor resolveMethod(IMethodDescriptor descriptor) throws CoreException {
426                        if (fContainer != null) {
427                                IReferenceTypeDescriptor type = descriptor.getEnclosingType();
428                                IApiTypeRoot classFile = fContainer.findTypeRoot(type.getQualifiedName());
429                                if(classFile != null) {
430                                        IApiType structure = classFile.getStructure();
431                                        IApiMethod[] methods = structure.getMethods();
432                                        for (int i = 0; i < methods.length; i++) {
433                                                IApiMethod method = methods[i];
434                                                if (descriptor.getName().equals(method.getName())) {
435                                                        String signature = method.getSignature();
436                                                        String descriptorSignature = descriptor.getSignature().replace('/', '.');
437                                                        if (Util.matchesSignatures(descriptorSignature, signature.replace('/', '.'))) {
438                                                                return descriptor.getEnclosingType().getMethod(method.getName(), signature);
439                                                        }
440                                                        String genericSignature = method.getGenericSignature();
441                                                        if (genericSignature != null) {
442                                                                if (Util.matchesSignatures(descriptorSignature, genericSignature.replace('/', '.'))) {
443                                                                        return descriptor.getEnclosingType().getMethod(method.getName(), signature);
444                                                                }
445                                                        }
446                                                }
447                                        }
448                                }
449                                throw new CoreException(new Status(IStatus.ERROR, ApiPlugin.PLUGIN_ID,
450                                        MessageFormat.format("Unable to resolve method signature: {0}", new String[]{descriptor.toString()}), null)); //$NON-NLS-1$
451                        }
452                        return descriptor;
453                }

509                InputStream inputStream = null;

513                } catch (FileNotFoundException e) {
514                        throw new CoreException(new Status(IStatus.ERROR, ApiPlugin.PLUGIN_ID,

522                } finally {

524                                try {

531                if(options == null) {

541                }
