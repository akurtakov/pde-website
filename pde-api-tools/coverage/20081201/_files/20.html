495         * Creates and returns an {@link IApiTypeContainer} at the specified path in
496         * this bundle, or <code>null</code> if the {@link IApiTypeContainer} does not
497         * exist. The path is the name (path) of entries specified by the
498         * <code>Bundle-ClassPath:</code> header.
499         * 
500         * @param path relative path to a class file container in this bundle
501         * @return {@link IApiTypeContainer} or <code>null</code>
502         * @exception IOException
503         */
504        protected IApiTypeContainer createApiTypeContainer(String path) throws IOException, CoreException {
505                File bundle = new File(fLocation);
506                if (bundle.isDirectory()) {
507                        // bundle is folder
508                        File entry = new File(bundle, path);
509                        if (entry.exists()) {
510                                if (entry.isFile()) {
511                                        return new ArchiveApiTypeContainer(this, entry.getCanonicalPath());
512                                } else {
513                                        return new DirectoryApiTypeContainer(this, entry.getCanonicalPath());
514                                }
515                        }
516                } else {
517                        // bundle is jar'd
518                        ZipFile zip = null;
519                        try {
520                                if (path.equals(".")) { //$NON-NLS-1$
521                                        return new ArchiveApiTypeContainer(this, fLocation);
522                                } else {
523                                        // TODO: use temporary space from OSGi if in a framework
524                                        zip = new ZipFile(fLocation);
525                                        ZipEntry entry = zip.getEntry(path);
526                                        if (entry != null) {
527                                                InputStream inputStream = null;
528                                                File tempFile;
529                                                FileOutputStream outputStream = null;
530                                                try {
531                                                        inputStream = zip.getInputStream(entry);
532                                                        tempFile = File.createTempFile("api", "tmp"); //$NON-NLS-1$ //$NON-NLS-2$
533                                                        tempFile.deleteOnExit();
534                                                        outputStream = new FileOutputStream(tempFile);
535                                                        byte[] bytes = new byte[8096];
536                                                        while (inputStream.available() > 0) {
537                                                                int read = inputStream.read(bytes);
538                                                                if (read > 0) {
539                                                                        outputStream.write(bytes, 0, read);
540                                                                }
541                                                        }
542                                                } finally {
543                                                        if (inputStream != null) {
544                                                                try {
545                                                                        inputStream.close();
546                                                                } catch(IOException e) {
547                                                                        ApiPlugin.log(e);
548                                                                }
549                                                        }
550                                                        if (outputStream != null) {
551                                                                try {
552                                                                        outputStream.close();
553                                                                } catch(IOException e) {
554                                                                        ApiPlugin.log(e);
555                                                                }
556                                                        }
557                                                }
558                                                return new ArchiveApiTypeContainer(this, tempFile.getCanonicalPath());
559                                        }
560                                }
561                        } finally {
562                                if (zip != null) {
563                                        zip.close();
564                                }
565                        }
566                }
567                return null;
568        }
