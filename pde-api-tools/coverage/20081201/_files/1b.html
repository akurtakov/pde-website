453        protected ManifestNode findNode(IElementDescriptor element, boolean write) {
454                IElementDescriptor[] path = element.getPath();
455                Map map = fPackageMap;
456                ManifestNode parentNode = null;
457                ManifestNode node = null;
458                for (int i = 0 ; i < path.length; i++) {
459                        IElementDescriptor current = path[i];
460                        parentNode = node;
461                        node = (ManifestNode) map.get(current);
462                        if (node == null) {
463                                if (write || (isInsertOnResolve(current))) {
464                                        node = createNode(parentNode, current);
465                                        if (node != null) {
466                                                map.put(current, node);
467                                        } else {
468                                                return null;
469                                        }
470                                } else if (!this.isExhaustive) {
471                                        return parentNode;
472                                } else {
473                                        return null;
474                                }
475                        }
476                        node = node.refresh();
477                        if (node != null) {
478                                map = node.children;
479                        }
480                }
481                return node;
482        }
